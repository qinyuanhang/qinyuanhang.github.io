Rotate List

Rotate List

用k = k%node的总数，在吧新的头，新的尾，链表的开头和结尾，在整合就可以

public class Solution {
    public ListNode rotateRight(ListNode head, int k) {

        if(head == null) return null;

        int tot = 1;
        ListNode head_1 = head;
        while(head_1.next != null){
            tot++;
            head_1 = head_1.next;
        }
        ListNode newEnd = head;
        k = k % tot;
        if(k == 0) return head;     
        k = tot - k;

        for(int i = 1; i < k; i++){
            newEnd = newEnd.next;
        }
        ListNode newHead = newEnd.next;
        head_1.next = head;
        newEnd.next = null;
        return newHead;
    }
}
